async function fetchYahoo(sym){
  const url=`https://query1.finance.yahoo.com/v8/finance/chart/${encodeURIComponent(sym)}?interval=1d&range=2d`;
  const proxy=`https://api.allorigins.win/get?url=${encodeURIComponent(url)}`;
  try{
    const res=await fetch(proxy,{signal:AbortSignal.timeout(7000)});
    const w=await res.json();if(!w.contents)return null;
    const data=JSON.parse(w.contents);
    const result=data?.chart?.result?.[0];if(!result)return null;
    const meta=result.meta;
    const price=meta.regularMarketPrice||meta.previousClose;
    if(!price||price<=0)return null;
    const prev=meta.chartPreviousClose||meta.previousClose||price;
    return{price:parseFloat(price),chg:parseFloat(((price-prev)/prev*100).toFixed(2))};
  }catch(e){return null}
}

function computeSignal(sym){
  const m=symMeta(sym),p=getP(sym);
  const atr=p*(0.003+Math.random()*0.005);const seed=(p*9973)%1;const rsi=28+seed*55;const bull=seed>0.42;
  let score=0;
  if(rsi<35)score+=1.5;else if(rsi<45)score+=0.5;else if(rsi>65)score-=1.5;else if(rsi>55)score-=0.5;
  score+=(bull?0.8:-0.8);score+=(Math.random()-.48)*1.8;
  const isBuy=score>0;const conf=Math.min(94,Math.round(54+Math.abs(score)/4.5*38));
  const buyV=Math.round(conf/100*150),sellV=150-buyV;
  const tpPts=Math.round(atr*1.5/m.pip),slPts=Math.max(1,Math.round(atr/m.pip));
  const tp=isBuy?p+tpPts*m.pip:p-tpPts*m.pip;const sl=isBuy?p-slPts*m.pip:p+slPts*m.pip;
  const rr=(tpPts/slPts).toFixed(2);const lot=Math.max(0.01,Math.min(10,Math.round(10000*0.01/(slPts*m.pv)*100)/100));
  const dir=conf<53?'WAIT':isBuy?'BUY':'SELL';
  const mtfs=['M5','M15','M30','H1','H4'];
  const mtfD=mtfs.map(()=>score>0?Math.random()>.35:Math.random()>.65);
  if(isBuy)mtfD[1]=true;else mtfD[1]=false;
  return{dir,conf,isBuy,buyV,sellV,entry:p,tp,sl,rr,tpPts,slPts,lot,mtfs,mtfD,atr,rsi:rsi.toFixed(1),real:hasReal(sym),label:m.l};
}
